create table diretoria (
codigo CHAR(3) not null primary key,
descricao char(10) not null
);

create table secao(
codigo char(3) not null primary key,
descricao char(10) not null,
diretoria char(3) not null references diretoria,
foreign key (diretoria)
references diretoria(codigo)
);

SELECT secao.codigo, secao.descricao, secao.diretoria FROM secao
INNER JOIN diretoria ON secao.diretoria=diretoria.codigo;
